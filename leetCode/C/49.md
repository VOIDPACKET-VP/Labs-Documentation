# Description
- Given an array of strings `strs`, group the anagrams together. You can return the answer in **any order**.

**Example 1:**
	**Input:** strs = ["eat","tea","tan","ate","nat","bat"]
	**Output:** [["bat"],["nat","tan"],["ate","eat","tea"]]

**Explanation:**
- There is no string in strs that can be rearranged to form `"bat"`.
- The strings `"nat"` and `"tan"` are anagrams as they can be rearranged to form each other.
- The strings `"ate"`, `"eat"`, and `"tea"` are anagrams as they can be rearranged to form each other.

**Example 2:**
	**Input:** strs = [""]
	**Output:** [[""]]

**Example 3:**
	**Input:** strs = ["a"]
	**Output:** [["a"]]

# Solution
- C
```
#include <stdlib.h>
#include <string.h>

typedef struct {
    char* original;
    char* sorted;
} Word;

// to compare char of each string
int compare(const void* a, const void* b){
    return (*(char*)a - *(char*)b);
}
void sortString(char* str){
    qsort(str, strlen(str), sizeof(char), compare);
}

// to compare strings to each other
int compareWords(const void* a, const void* b){
    Word* w1 = (Word*)a;
    Word* w2 = (Word*)b;
    return strcmp(w1->sorted, w2->sorted);
}

// main function
char*** groupAnagrams(char** strs, int strsSize, int* returnSize, int** returnColumnSizes) {
    if (strsSize == 0) {
        *returnSize = 0;
        return NULL;
    }

    int group = 0;
    if (strsSize > 0) group = 1;
    Word* words = malloc(strsSize * sizeof(Word)); // malloc instead of just 'Word words[strsSize]' because it's
    // better practice to use the heap instead of the stack
    for (int i = 0; i < strsSize; i++){
        words[i].original = strs[i];
        words[i].sorted = strdup(strs[i]); // used strdup() because it malloced in the heap and copies.
        sortString(words[i].sorted);
    }
    qsort(words, strsSize, sizeof(Word), compareWords);
    for (int i = 1; i < strsSize; i++){
        if (strcmp(words[i].sorted, words[i-1].sorted) != 0) group++;
    }

    *returnSize = group;
    *returnColumnSizes = malloc(sizeof(int) * (*returnSize));
    char*** result = malloc(sizeof(char**) * (*returnSize));

    int currentGroup = -1;
    int currentColumnSize = 0;
    for (int i = 0; i < strsSize; i++) {
        // If it's a new group
        if (i == 0 || strcmp(words[i].sorted, words[i-1].sorted) != 0) {
            currentGroup++;
            currentColumnSize = 0;
            // count how many words are in THIS specific group to malloc it
            int count = 0;
            for (int k = i; k < strsSize && strcmp(words[k].sorted, words[i].sorted) == 0; k++) {
                count++;
            }
            result[currentGroup] = malloc(sizeof(char*) * count);
            (*returnColumnSizes)[currentGroup] = count;
        }
        // Add the original string to the current group
        result[curentGroup][currentColumnSize++] = words[i].original;
    }
    return result;
}
```
